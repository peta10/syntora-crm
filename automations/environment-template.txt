# n8n Environment Configuration Template for Syntora Todo App
# Copy these variables to your n8n environment configuration

# n8n API Configuration (for MCP tools integration)

N8N_API_KEY=your-api-key-here

# Allow community packages to be used as AI tools

# Syntora App Integration
SYNTORA_API_BASE_URL=http://localhost:3000/api
SYNTORA_WEBHOOK_SECRET=your-webhook-secret-here

SUPABASE_SERVICE_ROLE_KEY=your-service-role-key
